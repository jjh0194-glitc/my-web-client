<script>
    // 1. 배포된 서버 주소로 반드시 변경하세요
    const SERVER_URL = 'https://my-web-server-5qln.onrender.com';

    fetch(`${SERVER_URL}/api/wedding-info`)
        .then(res => res.json())
        .then(data => {
            document.getElementById('title').innerText = data.title;
            document.getElementById('date').innerText = data.date;
            document.getElementById('location').innerText = data.location;
            document.getElementById('account').innerText = data.account;
        })
        .catch(err => {
            document.getElementById('title').innerText = "정보를 불러올 수 없습니다.";
        });

    // 계좌번호 복사 기능
    function copyAccount() {
        const text = document.getElementById('account').innerText;
        navigator.clipboard.writeText(text).then(() => {
            alert("계좌번호가 복사되었습니다.");
        });
    }
